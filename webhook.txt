#include <iostream>
#include <random>
#include <string>

/**
 * Generates a random binary string of a specified length.
 * @param length The desired length of the binary string.
 * @return A string containing random '0's and '1's.
 */
std::string generate_random_binary_string(int length) {
    // Obtain a random number from hardware (if available) to seed the generator
    std::random_device rd; 
    
    // Seed the Mersenne Twister engine
    std::mt19937 eng(rd()); 
    
    // Define the range for the distribution: 0 or 1
    std::uniform_int_distribution<> distr(0, 1); 

    std::string binary_code = "";
    for (int n = 0; n < length; ++n) {
        // Generate a random 0 or 1 and append it to the string
        binary_code += std::to_string(distr(eng)); 
    }
    return binary_code;
}

int main() {
    int code_length = 40; // Example length
    std::string random_binary = generate_random_binary_string(code_length);
    
    std::cout << "Generated random binary code (" << code_length << " bits):" << std::endl;
    std::cout << random_binary << std::endl;
    
    return 0;
}
